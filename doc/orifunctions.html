library(MASS)

set.seed(123)
p <- 5
n <- 400
r <- 5
w <- runif(p)
sigma2 <- runif(p)
U0 <- matrix(runif(p*r),nrow=p)
U <- qr.Q(qr(diag(w)%*%U0))
sigma0 <- U%*%diag(1:r)%*%t(U)
x <- mvrnorm(n,mu=rep(0,r),Sigma = sigma0)
e <- mvrnorm(n,mu=rep(0,r),Sigma = diag(sigma2*sigma2))
y <- x+e
for(i in 1:r){
  y[,i]=y[,i]-mean(y[,i])
}
hat <- t(y)%*%y/(n-1)
U0 <- Heteropca(hat,2,500)
1-svd(U[-(1:2),]%*%U0)$d[1]
